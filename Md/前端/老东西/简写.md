### 一.三元运算

```javascript
const flag = false
flag ? console.log('ok') : console.log('no')//输出no
// 如果只有一半结果，则把没有结果的定义为null，如 flag ? console.log('ok') : null
```

### 二.短路运算	

```javascript
const test = str => console.log(str)
const flag = true
flag && test('lll')
```

```javascript
// 原写法
// const test = flag => {
// 	if (flag) {
// 		return flag
// 	} else {
// 		return 'no'
// 	}
// }
// 简化后
const test = flag => flag || 'no'
console.log(test(flag)) //输出no
```

### 三.非分支策略

```javascript
// 原 switch 写法
// const lists = key => {
// 	switch (key) {
// 		case 'name':
// 			return 'jack';
// 			break;
// 		case 'age':
// 			return 18;
// 			break;
// 		case 'job':
// 			return web;
// 			break;
// 		default:
// 			return 'not found'
// 	}
// }
// 新写法
const lists = key => ({
	'name': 'jack',
	'age': 18,
	'job': 'web'
})[key] || 'not found'

console.log(lists('name'))//输出 jack
```
