## 具名插槽组件

```tsx
function List<T extends unknown[]>(props: { list: T; children: (item: T[0]) => React.ReactElement }) {
  return <ul>{props.list.map((item) => props.children(item))}</ul>;
}
function Item(props: { v: number }) {
  return <li>{props.v}</li>;
}
// <List list={[-1, -2, 0, 1, 2]}>{(item) => <Item key={item} v={item} />}</List>
```
